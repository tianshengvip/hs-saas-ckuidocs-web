快速开始 #

本节将介绍如何在项目中使用 CaoKunTSVue3UI

用法 #

// 在main.ts中导入 使用

import CaokunTsvue3UI from 'caokun-tsvue3-ui';
import '../node_modules/caokun-tsvue3-ui/lib/style.css';
const app = createApp(App);

app.use(CaokunTsvue3UI).mount('#app');
